for ($i=1; $i<=2; $i++) {               // 表格有2列 
    echo "<tr>";
    $fontcolor=($i==1)?"white":"black";     //文字顏色
    $bgcolor=($i==1)?"orange":"#FFDAB9";    //背景顏色

    for ($j=1; $j<=4; $j++) {         //每列4行
        echo "<td height='30' bgcolor=$bgcolor>";
        echo "<font color=$fontcolor>";
        $x=($i==1)?$j+1:$j+5;
        for($y=1;$y<10;$y++){
            $product=$x*$y;
            echo $x."*".$y."=".$product."<br>";
        }
        echo "</font>";
        echo"</td>";
    }
    echo "</tr>";
 }
